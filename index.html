        function renderTabs() {

            const container = document.getElementById('day-tabs-container');

            itineraryData.forEach((data, index) => {

                const btn = document.createElement('button');

                btn.className = `day-tab whitespace-nowrap px-5 py-2 rounded-full border border-stone-300 font-medium text-sm transition focus:outline-none ${index === 0 ? 'active shadow-md' : 'bg-white text-slate-600 hover:bg-stone-100'}`;

                btn.innerText = `Day ${data.day}`;

                btn.onclick = () => selectDay(index, btn);

                container.appendChild(btn);

            });

        }


        function selectDay(index, clickedBtn) {

            document.querySelectorAll('.day-tab').forEach(btn => {

                btn.classList.remove('active', 'shadow-md');

            });

            clickedBtn.classList.remove('bg-white', 'text-slate-600');

            clickedBtn.classList.add('active', 'shadow-md');


            const data = itineraryData[index];

            contentDiv.innerHTML = `
